## Hoisting

Before:

```
var i;

function moreMath() {
	console.log('moremath!!');
};

console.log('line 22');

var myName = 'Kris';
if(myName == 'Kris') {
	myCat = 'Spot';
}
var myCat = 'Luna';

(function anotherCar() {
	var x = 10;
	console.log('function ran!');
})();

return;

var i;
```

After:
```
var i = undefined;
var myName = undefined;
var myCat = undefined;

function moreMath() {
	console.log('moremath!!');
};

var myName = 'Kris';	//	From undefined
var myCat = 'Luna';		// 	From undefined

return;		//	Ends execution

var i;
```
